create table Employee
(
    Id int,
    EmployeeId character varying(10) UNIQUE NOT NULL ,
    FullName character varying(100)  NOT NULL,
    BirthDate date NOT NULL,
    Address character varying(500),
    CONSTRAINT "Employee_pkey" PRIMARY KEY (Id)
)

CREATE TABLE positionhistory
(
    Id Int,
    Posid character varying(10) NOT NULL,
    PosTitle character varying(100) NOT NULL,
    Employeeid character varying(10) NOT NULL,
    StartDate Date NOT NULL,
    EndDate Date NOT NULL,
    CONSTRAINT "PositionHistory_pkey" PRIMARY KEY (Id)
)

CREATE OR REPLACE VIEW v_employee_position AS
 SELECT a.employeeid,
    a.fullname,
    a.birthdate,
    a.address,
    b.postitle
   FROM employee a
     LEFT JOIN positionhistory b ON a.employeeid = b.employeeid AND CURRENT_DATE >= b.startdate AND CURRENT_DATE <= b.enddate;
